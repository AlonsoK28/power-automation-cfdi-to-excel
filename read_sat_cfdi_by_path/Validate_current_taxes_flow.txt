FUNCTION Validate_current_taxes GLOBAL
    SET myIndex2 TO 0
    IF ImpuestosArr.Count >= 1 THEN
        LOOP FOREACH Current_Tax IN Impuestos_TasaOCuotaArr
            # Validar "Base" e "Importe" de los Impuestos del CFDI actual
            SWITCH Current_Tax.InnerText
                CASE = 0.000000
                    Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: Impuestos_BaseArr[myIndex2].innerText Column: Column_Impuestos_Base_0 Row: myIndex + 1
                    Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: Impuestos_ImporteArr[myIndex2].innerText Column: Column_Impuestos_Importe_0 Row: myIndex + 1
                CASE = 0.080000
                    Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: Impuestos_BaseArr[myIndex2].innerText Column: Column_Impuestos_Base_8 Row: myIndex + 1
                    Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: Impuestos_ImporteArr[myIndex2].innerText Column: Column_Impuestos_Importe_8 Row: myIndex + 1
                CASE = 0.160000
                    Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: Impuestos_BaseArr[myIndex2].innerText Column: Column_Impuestos_Base_16 Row: myIndex + 1
                    Excel.WriteToExcel.WriteCell Instance: ExcelInstance Value: Impuestos_ImporteArr[myIndex2].innerText Column: Column_Impuestos_Importe_16 Row: myIndex + 1
            END
            Variables.IncreaseVariable Value: myIndex2 IncrementValue: 1
        END
    END
END FUNCTION
